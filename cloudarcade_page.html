<script>
$(document).ready(function() {
// Set timezone to Ho Chi Minh City
moment.tz.setDefault("Asia/Ho_Chi_Minh");

function setDateRange(startDate, endDate) {
$('#startDate').val(startDate.format('YYYY-MM-DD'));
$('#endDate').val(endDate.format('YYYY-MM-DD'));
}

// Set default to today
setDateRange(moment().startOf('day'), moment().endOf('day'));

$('#todayBtn').click(function() {
setDateRange(moment().startOf('day'), moment().endOf('day'));
});

$('#last7DaysBtn').click(function() {
setDateRange(moment().subtract(6, 'days').startOf('day'), moment().endOf('day'));
});

$('#last28DaysBtn').click(function() {
setDateRange(moment().subtract(27, 'days').startOf('day'), moment().endOf('day'));
});

$('#fetchData').click(function() {
const startDate = $('#startDate').val();
const endDate = $('#endDate').val();
const regenerate = $('#regenerateCheck').is(':checked');
$.ajax({
url: 'api/api_cloudarcade.php',
method: 'GET',
data: {
startDate: startDate,
endDate: endDate,
regenerate: regenerate
},
dataType: 'json',
success: function(response) {
displayJsonData(response);
displayGames(response.games);
},
error: function(xhr, status, error) {
alert('Error fetching game data: ' + error);
}
});
});

function displayJsonData(data) {
$('#jsonData').val(JSON.stringify(data, null, 2));
}

function displayGames(games) {
const container = $('#gameContainer');
container.empty();
games.forEach(function(game) {
const gameCard = `
<div class="col-md-4 game-card">
<div class="card">
<img src="${game.game_thumbnail || 'placeholder.jpg'}" class="card-img-top game-image" alt="${game.name}">
<div class="card-body">
<h5 class="card-title">${game.name}</h5>
<p class="card-text">${game.game_desc || 'No description available.'}</p>
<p><strong>Instructions:</strong> ${game.game_instruction || 'No instructions available.'}</p>
<p><strong>Category:</strong> ${game.category || 'Uncategorized'}</p>
<p><strong>Generated Date:</strong> ${game.generated_date}</p>
<a href="${game.gameplay_url}" class="btn btn-primary" target="_blank">Play Game</a>
</div>
</div>
</div>
`;
container.append(gameCard);
});
}
});

}
