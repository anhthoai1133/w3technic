<script>
$(document).ready(function() {
var table = $('#extensions').DataTable({
processing: true,
serverSide: false,
ajax: {
url: "extension_api.php",
type: "POST",
data: { action: 'READ_EXTENSIONS' }
},
columns: [
{ data: "id" },
{ data: "name" },
{ data: "version" },
{ data: "status" },
{ data: "last_updated" },
{ data: "website_url" },
{ data: "url" },
{ data: "users" },
{
data: null,
render: function(data, type, row) {
return '<button class="btn btn-primary btn-sm edit-extension" data-id="' + row.id + '">Edit</button> ' +
'<button class="btn btn-danger btn-sm delete-extension" data-id="' + row.id + '">Delete</button>';
},
orderable: false
}
]
});

// Load websites into dropdown
$.ajax({
url: 'extension_api.php',
type: 'POST',
data: { action: 'GET_WEBSITES' },
dataType: 'json',
success: function(response) {
if (response.status === 'success') {
var select = $('#website_id');
$.each(response.data, function(i, website) {
select.append($('<option></option>').val(website.id).text(website.url));
});
} else {
console.error('Error loading websites:', response.message);
}
},
error: function() {
console.error('An error occurred while loading websites.');
}
});

$('#addNew').on('click', function() {
$('#modalTitle').text('Add New Extension');
$('#extensionForm')[0].reset();
$('#extensionId').val('');
$('#extensionModal').modal('show');
});

$('#extensionForm').on('submit', function(e) {
e.preventDefault();
var action = $('#extensionId').val() ? 'UPDATE_EXTENSION' : 'CREATE_EXTENSION';
var data = {
action: action,
id: $('#extensionId').val(),
name: $('#name').val(),
version: $('#version').val(),
status: $('#status').val(),
url: $('#url').val(),
website_id: $('#website_id').val(),
users: $('#users').val()
};

$.ajax({
url: 'extension_api.php',
type: 'POST',
data: data,
dataType: 'json',
success: function(response) {
if (response.status === 'success') {
$('#extensionModal').modal('hide');
table.ajax.reload();
alert(response.message);
} else {
alert('Error: ' + response.message);
}
},
error: function() {
alert('An error occurred while submitting the form.');
}
});
});

$('#extensions').on('click', '.edit-extension', function() {
var id = $(this).data('id');
$.ajax({
url: 'extension_api.php',
type: 'POST',
data: {
action: 'READ_EXTENSION',
id: id
},
dataType: 'json',
success: function(response) {
if (response.status === 'success') {
$('#modalTitle').text('Edit Extension');
$('#extensionId').val(response.data.id);
$('#name').val(response.data.name);
$('#version').val(response.data.version);
$('#status').val(response.data.status);
$('#url').val(response.data.url);
$('#website_id').val(response.data.website_id);
$('#users').val(response.data.users);
$('#extensionModal').modal('show');
} else {
alert('Error: ' + response.message);
}
},
error: function() {
alert('An error occurred while fetching the extension details.');
}
});
});

$('#extensions').on('click', '.delete-extension', function() {
var id = $(this).data('id');
if (confirm('Are you sure you want to delete this extension?')) {
$.ajax({
url: 'extension_api.php',
type: 'POST',
data: {
action: 'DELETE_EXTENSION',
id: id
},
dataType: 'json',
success: function(response) {
if (response.status === 'success') {
table.ajax.reload();
alert('Extension deleted successfully');
} else {
alert('Error: ' + response.message);
}
},
error: function() {
alert('An error occurred while deleting the extension.');
}
});
}
});
});

$(document).ready(function() {
function toggleSidebar() {
if ($(window).width() < 768) {
$('#sidebarMenu').toggleClass('show');
}
}

$('.navbar-toggler').on('click', function(e) {
e.preventDefault();
toggleSidebar();
});

$(document).on('click', function(e) {
if ($(window).width() < 768) {
if (!$(e.target).closest('#sidebarMenu').length && !$(e.target).closest('.navbar-toggler').length) {
$('#sidebarMenu').removeClass('show');
}
}
});

$('#sidebarMenu').on('click', function(e) {
e.stopPropagation();
});

$(window).on('resize', function() {
if ($(window).width() >= 768) {
$('#sidebarMenu').removeClass('show');
}
});
});
